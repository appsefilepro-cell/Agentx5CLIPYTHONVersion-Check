Why Replit (again) & reality check

Replit gives you a working Linux environment, Python, and a public preview URL — great for testing and orchestration.

Heavy local LLMs (Ollama, local GGUF models) and GPU workloads cannot realistically run in Replit — use a cloud VM or a local machine for that later.

For now we’ll run a fully offline/simulated trading pipeline that behaves like a real bot (price feed simulation, logging, risk controls, rollback, and a small UI). Later you can swap the simulator for CCXT/live exchange code.

2) backend/trading_agent.py — SIMULATOR (sandbox)
import asyncio, random, sqlite3, datetime, os
os.makedirs("logs", exist_ok=True)

class TradingAgent:
    def __init__(self):
        self.symbol="BTC/USD"
        self.pos=0.0
        self.entry=0.0

    async def price_stream(self):
        price=30000.0
        while True:
            # simulate small random walk
            price += random.uniform(-150,150)
            yield round(price,2)
            await asyncio.sleep(1)

    async def start(self):
        async for p in self.price_stream():
            await self.on_price(p)

    async def on_price(self, price):
        # very simple strategy: buy if price drops > 0.3% from last, sell if +0.5% gain
        conn=sqlite3.connect("database/trading.db")
        c=conn.cursor()
        bal=c.execute("SELECT amount FROM balance").fetchone()[0]
        action=None
        if self.pos==0 and random.random()<0.02:
            # buy small
            size = min(0.01, bal/price)
            if size*price < bal:
                self.pos += size; self.entry=price; bal -= size*price; action="BUY"
                c.execute("INSERT INTO trades(symbol,side,price,size,pnl,ts) VALUES (?,?,?,?,?,?)",
                          (self.symbol,"BUY",price,size,0.0,str(datetime.datetime.utcnow())))
        elif self.pos>0 and (price - self.entry)/self.entry > 0.005:
            # take profit
            pnl = (price - self.entry) * self.pos
            bal += price*self.pos
            c.execute("INSERT INTO trades(symbol,side,price,size,pnl,ts) VALUES (?,?,?,?,?,?)",
                      (self.symbol,"SELL",price,self.pos,pnl,str(datetime.datetime.utcnow())))
            self.pos=0; self.entry=0; action="SELL"

        c.execute("UPDATE balance SET amount = ?", (bal,))
        conn.commit(); conn.close()
        # log
        with open("logs/trades.log","a") as f:
            f.write(f"{datetime.datetime.utcnow()} price={price} action={action} pos={self.pos}\n")

5) frontend/simple_ui.py (FastAPI minimal UI)
from fastapi import FastAPI
import sqlite3, json
app=FastAPI()
@app.get("/balance")
def balance():
    conn=sqlite3.connect("database/trading.db")
    b=conn.execute("SELECT amount FROM balance").fetchone()[0]
    trades=[dict(zip([c[0] for c in conn.execute("PRAGMA table_info(trades)").fetchall()], row)) for row in conn.execute("SELECT * FROM trades").fetchall()]
    conn.close()
    return {"balance":b,"trades":trades}
